Alur Kerja Sistem (Flowchart)

graph TD
    %% 1. Autentikasi Admin
    Start((Mulai)) --> Public[Halaman Depan / Scan QR]
    Public --> Login{Admin Login?}
    Login -- Ya --> Auth[Verifikasi Akun]
    Auth --> Dashboard[Dashboard Admin]
    Login -- Tidak --> QRScan[Scan QR Kode Ruangan]

    %% 2. Alur Pengelolaan Aset
    Dashboard --> Menu{Pilih Menu}
    
    Menu --> Master[Master Data]
    Master --> CRUD_Master[Kelola Lokasi/Ruangan/Klasifikasi]
    CRUD_Master --> Dashboard

    Menu --> Aset[Inventaris Aset]
    Aset --> CRUD_Aset[Tambah/Edit/Hapus Aset]
    CRUD_Aset --> GenQR[Generate QR Code]
    GenQR --> Dashboard

    Menu --> Mutasi[Mutasi Aset]
    Mutasi --> SelectAset[Pilih Aset & Ruangan Tujuan]
    SelectAset --> SaveMutasi[Simpan Histori Mutasi]
    SaveMutasi --> UpdateLoc[Update Lokasi Aset Otomatis]
    UpdateLoc --> Dashboard

    Menu --> Report[Laporan]
    Report --> Export[Export PDF / Excel]
    Export --> Dashboard

    %% 3. Alur QR Scan (Public User)
    QRScan --> ViewRoom[Tampil Daftar Aset Ruangan]
    ViewRoom --> Detail[Lihat Detail Aset]
    Detail --> End((Selesai))

    classDef process fill:#fff,stroke:#333,stroke-width:2px;
    classDef decision fill:#f9f9f9,stroke:#333,stroke-width:2px;
    classDef startend fill:#000,color:#fff,stroke-width:2px;

    class Start,End startend;
    class Dashboard,Public,Auth,CRUD_Master,CRUD_Aset,GenQR,SelectAset,SaveMutasi,UpdateLoc,Export,ViewRoom,Detail process;
    class Login,Menu decision;
