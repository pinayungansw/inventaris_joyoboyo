    MASTER_LOKASI {
        bigint id PK "Primary Key"
        varchar nama_lokasi "Nama Lokasi/Kantor"
    }

    MASTER_RUANGAN {
        bigint id PK "Primary Key"
        bigint lokasi_id FK "Foreign Key ke Master Lokasi"
        varchar kode_ruangan "Kode Unik Ruangan"
        varchar nama_ruangan "Nama Ruangan"
    }

    MASTER_KLASIFIKASI {
        bigint id PK "Primary Key"
        varchar nama_klasifikasi "Kelompok Klasifikasi"
    }

    MASTER_JENIS_BARANG {
        bigint id PK "Primary Key"
        bigint klasifikasi_id FK "Foreign Key ke Master Klasifikasi"
        varchar kode_jenis "Kode Unik Jenis"
        varchar nama_jenis "Nama Jenis Barang"
    }

    ASET {
        bigint id PK "Primary Key"
        varchar kode_aset UK "Unique Key/Barcode"
        bigint jenis_barang_id FK "Foreign Key ke Master Jenis"
        bigint ruangan_id FK "Foreign Key ke Master Ruangan"
        varchar merk "Merk Barang"
        text spesifikasi "Detail Spesifikasi"
        varchar tahun_pembelian "Tahun Pengadaan"
        enum kondisi "Status: Baik/Rusak/Layak"
        integer jumlah "Kuantitas"
        varchar nomor_seri_inventaris "No Seri Fisik"
        text keterangan "Catatan Tambahan"
    }

    MUTASI_ASET {
        bigint id PK "Primary Key"
        bigint aset_id FK "Foreign Key ke Aset"
        bigint dari_ruangan_id FK "Foreign Key (Asal)"
        bigint ke_ruangan_id FK "Foreign Key (Tujuan)"
        date tanggal_mutasi "Waktu Mutasi"
        text keterangan "Alasan Mutasi"
    }

    USERS {
        bigint id PK "Primary Key"
        varchar name "Nama Lengkap"
        varchar email UK "Email Login"
        varchar password "Hash Password"
    }
